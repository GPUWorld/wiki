PS1

# extension of \u@\h:\w\$
# \u: username
# \w: working directory
$ export PS1=`echo -ne "\033[0;34m\u@\h:\033[0;36m\w\033[0;34m\$\033[0;37m "`

PROMPT_COMMAND

# run just before bash gives you a prompt
$ export PROMPT_COMMAND='echo -ne "\033[0;${USER}@${HOSTNAME}: ${PWD}\007"'

ignore consecutive ^D hits

$ export IGNOREEOF=2

clear out history quickly

$ export HISTSIZE=0

  • $PATH
  • $CDPATH
  • $MANPATH

TMOUT

$ export TMOUT=600
